```python
"""Enhanced base class eliminating duplication."""
from abc import ABC, abstractmethod
from typing import Optional
from ase.calculators.calculator import Calculator

class BasePotential(Calculator, ABC):
    """Enhanced base class with common ML potential functionality."""
    
    implemented_properties = ["energy", "forces"]
    
    def __init__(self, model_name=None, device=None, **kwargs):
        super().__init__(**kwargs)
        self.model_name = self._resolve_model_name(model_name)
        self.device = self._resolve_device(device)
        self._calculator = None
        
    def _resolve_device(self, device):
        """Standardized device resolution logic."""
        if device is None:
            try:
                import torch
                return "cuda" if torch.cuda.is_available() else "cpu"
            except ImportError:
                return "cpu"
        return device
        
    def _handle_precision_errors(self, error, atoms, properties):
        """Common precision error handling for PyTorch models."""
        error_msg = str(error)
        if "expected scalar type Double but found Float" in error_msg:
            # Handle dtype mismatch
            if hasattr(self, '_calculator') and hasattr(self._calculator, 'model'):
                if hasattr(self._calculator.model, 'double'):
                    self._calculator.model.double()
                    return True
        return False
        
    @abstractmethod  
    def _load_model(self):
        """Subclasses implement specific model loading."""
        pass
''')
```

```python
create_file("qme/core/results.py", '''"""
Structured result objects for QME operations.

Replaces dictionary-based results with typed, structured objects
while maintaining backward compatibility.
"""
from dataclasses import dataclass, asdict
from typing import Optional, Dict, Any
from ase import Atoms
import time

@dataclass
class OptimizationResult:
    """Result object for optimization operations."""
    optimized_atoms: Atoms
    final_energy: float
    steps_taken: int
    converged: bool
    optimization_time: float
    initial_energy: Optional[float] = None
    optimizer_name: Optional[str] = None
    
    def save(self, filename: str) -> None:
        """Save optimized structure to file."""
        from ase.io import write
        write(filename, self.optimized_atoms)
        
    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for backward compatibility."""
        result = asdict(self)
        # Handle ASE Atoms object specially
        result['optimized_atoms'] = self.optimized_atoms
        return result
        
    @property  
    def energy_change(self) -> Optional[float]:
        """Calculate energy change if initial energy available."""
        if self.initial_energy is not None:
            return self.final_energy - self.initial_energy
        return None

@dataclass
class FrequencyResult:
    """Result object for frequency calculations."""
    frequencies: list
    normal_modes: Optional[Any] = None
    zero_point_energy: Optional[float] = None
    enthalpy: Optional[float] = None
    free_energy: Optional[float] = None
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary for backward compatibility."""
        return asdict(self)
''')
```

```python
        logfile: Path to optimization log file. Optional.
        trajectory: Path to save optimization trajectory. Optional.
        constraints: ASE constraints to apply during optimization. Optional.
        **kwargs: Additional arguments passed to optimizer.
        
    Returns:
        OptimizationResult: Structured result containing:
            - optimized_atoms: Final optimized ASE Atoms object
            - final_energy: Converged energy in eV  
            - steps_taken: Number of optimization steps performed
            - converged: Whether optimization converged to fmax
            - optimization_time: Wall-clock optimization time in seconds
            - initial_energy: Starting energy before optimization
            
    Raises:
        StructureNotLoadedError: If no structure loaded and atoms=None
        ValueError: If optimizer not recognized or parameters invalid
        RuntimeError: If optimization encounters calculator errors
        
    Examples:
        Basic optimization:
        
        >>> qme = QMEOptimizer(backend='aimnet2')
        >>> qme.load_structure('molecule.xyz')
        >>> result = qme.optimize_minimum(fmax=0.005, steps=500)
        >>> print(f"Final energy: {result.final_energy:.6f} eV")
        >>> result.save('optimized.xyz')
        
        With constraints:
        
        >>> result = qme.optimize_minimum(
        ...     constraints=[FixAtoms(indices=[0, 1, 2])],
        ...     optimizer='BFGS',
        ...     fmax=0.01
        ... )
        >>> print(f"Converged: {result.converged}")
        
    Note:
        - Uses the calculator backend specified during QMEOptimizer initialization
        - Constraint parsing is handled automatically for dict-style constraints
        - Optimization trajectory saved if trajectory parameter provided
    """''')
```

```python
# 1. Create plugin base classes
create_file("qme/core/plugins.py", '''"""
Plugin architecture for QME backend extensibility.

Extends the existing calculator registry to support dynamic backend
registration and plugin-based architecture.
"""
from abc import ABC, abstractmethod
from typing import Dict, Any, List, Optional, Type
from qme.types.protocols import CalculatorProtocol, BackendProtocol

class BackendPlugin(ABC):
    """Base class for QME backend plugins."""
    
    @property
    @abstractmethod
    def name(self) -> str:
        """Unique backend name."""
        pass
    
    @abstractmethod
    def get_calculator(self, model_name: str, **kwargs) -> CalculatorProtocol:
        """Create calculator instance."""
        pass
    
    @abstractmethod
    def get_default_model(self) -> str:
        """Get default model name for this backend."""
        pass
        
    @abstractmethod
    def check_dependencies(self) -> bool:
        """Check if required dependencies are available."""
        pass
        
    @abstractmethod
    def get_supported_properties(self) -> List[str]:
        """Get list of supported calculation properties."""
        pass

class PluginRegistry:
    """Registry for managing backend plugins."""
    
    def __init__(self):
        self._plugins: Dict[str, BackendPlugin] = {}
        self._builtin_backends = {
            'mock', 'uma', 'aimnet2', 'so3lr', 'mace'
        }
        
    def register_plugin(self, plugin: BackendPlugin) -> None:
        """Register a new backend plugin."""
        if plugin.name in self._builtin_backends:
            raise ValueError(f"Cannot override builtin backend: {plugin.name}")
        self._plugins[plugin.name] = plugin
        
    def unregister_plugin(self, name: str) -> None:
        """Unregister a backend plugin.""" 
        if name in self._plugins:
            del self._plugins[name]
            
    def get_available_backends(self) -> List[str]:
        """Get list of available backends including plugins."""
        from qme.core.registry import calculator_registry
        builtin = calculator_registry.get_available_backends()
        plugin = [name for name, plugin in self._plugins.items() 
                 if plugin.check_dependencies()]
        return builtin + plugin
        
    def create_calculator(self, backend: str, **kwargs) -> CalculatorProtocol:
        """Create calculator, checking plugins first."""
        if backend in self._plugins:
            plugin = self._plugins[backend]
            if plugin.check_dependencies():
                return plugin.get_calculator(**kwargs)
            else:
                raise RuntimeError(f"Plugin {backend} dependencies not satisfied")
                
        # Fall back to builtin registry
        from qme.core.registry import calculator_registry  
        return calculator_registry.create_calculator(backend=backend, **kwargs)

# Global plugin registry
plugin_registry = PluginRegistry()
''')
```
